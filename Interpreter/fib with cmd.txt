.data
fn 0 // Инициализированная переменная fn = 0
f1 0 // Инициализированная переменная f1 = 0
f2 1 // Инициализированная переменная f2 = 1
n 21 // Инициализированная переменная n = 21 - То, который меняется

.code
mov eax 2 // Счётчик eax равный 2

for_start // Начало цикла
    mov ebx f1 // В ebx сохраняется значение переменной f1
    add f1 f2 // К переменной f1 прибавляем значение переменной f2
    mov fn f1 // Присваиваем переменной fn значение переменной f1
    mov f2 ebx // Присваиваем переменной f2 значение переменной ebx
    cmp eax n // Сравнение eax и n
    jge for_end // Если eax больше или равно, то переход к for_end (Завершение цикла)
    add eax 1 // Иначе, +1 к счётчику
    jmp for_start // Переход к for_start (Начало цикла)
for_end // Конец цикла

output fn // Вывести fn
end // Завершаем процесс
